class Solution {
    public int[] twoSum(int[] nums, int target) {
        
        //we can put all elements in a hashmap and then find the difference of target and each elements 
        //look it up in the hashmap , if it exists then return the indices 
        HashMap<Integer,Integer> map=new HashMap<>();
        int[] a=new int[2];
        for(int i=0;i<nums.length;i++){
            if(!map.containsKey(nums[i]))
            map.put(nums[i],i);
        }
        for(int i=0;i<nums.length;i++){
            int diff=target-nums[i];
            if(map.containsKey(diff) && map.get(diff) != i){
                a[0]=i;
                a[1]=map.get(diff);
                break;
            }
        }
        return a;
    }
}
