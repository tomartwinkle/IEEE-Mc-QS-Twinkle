class Solution {
    static boolean isPrime(int i){
        if(i<=1) return false;
        for(int j=2;j<=Math.sqrt(i);j++){
            if(i%j==0) return false;
        }
        return true;
    }
    public boolean isUgly(int n) {
        if (n <= 0) return false;
        if (n == 1) return true;

        for (int i = 2; i <= n / 2; i++) {
            if (n % i == 0&&isPrime(i)) {
                if (i != 2 && i != 3 && i != 5) return false;
                
            }
        }
        if(isPrime(n)&&n!=2&&n!=3&&n!=5) return false;
        return true;
    }
}
